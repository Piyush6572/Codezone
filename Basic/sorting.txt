#include<stdio.h>
#include<string.h>
int main()
{
    char str[10][50], t[50];
    int i,j,k,n;
    printf("Enter the no. of strings to be sorted: ");
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        printf("Enter string %d: ",i+1);
        scanf("%s",str[i]);
    }
    for(i=0;i<n-1;i++)
    {
        for(j=i+1;j<n;j++)
        {
            if(strcmp(str[i],str[j])>0)
            {
                strcpy(t,str[i]);
                strcpy(str[i],str[j]);
                strcpy(str[j],t);
            }
        }
    }
    for(i=0;i<n;i++)
    printf("%s\n",str[i]);
    return 0;
}
