#include<stdio.h>
int main()
{
    int a,b,c=0;
    printf("enter a number:");
    scanf("%d",&a);
    b=a;
    while (b>0)
    {
        c=c*10 +(b%10);
        b=b/10;
    }
    if(c==a)
        printf("it is a palindrone");
    else
        printf("it is not a palindrone");
    return 0;
}
