#include<stdio.h>
int main()
{
    int n,i,j,k=11,num=0;
    printf ("enter the no. of lines: ");
    scanf ("%d",&n);

    for(i=0;i<n;i++)
    {
        for(j=0;j<n-i-1;j++)
        printf ("\t");

        for (j=0;j<i;j++)
        {
            num=k-j;
            printf ("%d\t",num);
        }
        printf ("%d\t",k-j);

        for (j=i-1;j>=0;j--)
        {
            num=k-j;
            printf ("%d\t",num);
        }
        printf ("\n");
    }
    return 0;
}
