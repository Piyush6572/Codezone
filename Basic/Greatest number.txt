#include<stdio.h>
int main()
{
    int a,b,c,d;
    printf("Enter the three numbers:");
    scanf("%d%d%d",&a,&b,&c);
    d= (a>b)?((a>c)?a:c):((b>c)?b:c);
    printf("\n %d",d);
    return 0;
}
