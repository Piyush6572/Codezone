#include<stdio.h>
int main()
{
    int i=2,sum=0;
    while (i<150)
    {
        printf("%d\t",i);
        i+=3;
        if (i%10==0)
        sum+=i;
    }
    printf("\nsum=%d",sum);
    return 0;
}
