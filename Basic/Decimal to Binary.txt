#include<stdio.h>

char digits[16];

int conbase(long n, int base)
{
    char fnl[100];
    int count=0,i;
    while(n>0)
    {
        fnl[count++]=digits[n%base];
        n/=base;
    }
    for(i=count-1;i>=0;--i)
        printf ("%c",fnl[i]);
    printf ("\n");
}

int main()
{
    long n;
    int base,i;
    for(i=0;i<=9;++i)
        digits[i]='0'+i;
    for(i=0;i<6;++i)
        digits[10+i]='A'+i;

    printf ("Enter decimal no. and base: ");
    scanf ("%ld%d",&n,&base);
    conbase(n,base);
    return 0;
}
