#include<stdio.h>
int func(int n)
{
    while(n>0)
    {if(n==1)
        return 1;
    else
        return(n+func(n-1));
    }
}
int main()
{
    int N,s;
    printf("Enter the value of N:");
    scanf("%d",&N);
    s=2*func(N);
    printf("Sum is %d",s);
    return 0;
}
